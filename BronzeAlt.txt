int r1, r2;
int flag = 0;
const int trigPin = 13;
const int echoPin = 12;
long duration;
int distance;
bool obstacleClear = false;
unsigned long StartTimeG = 0;
unsigned long ElapsedTimeG = 0;
unsigned long parkTriggerTime = 0;

void setup() {
    pinMode(A1, INPUT);
    pinMode(A3, INPUT);
    pinMode(5, OUTPUT);
    pinMode(6, OUTPUT);
    pinMode(7, OUTPUT);
    pinMode(8, OUTPUT);
    pinMode(trigPin, OUTPUT);
    pinMode(echoPin, INPUT);
    Serial.begin(9600);
}

void stopBuggy() {
    digitalWrite(5, LOW);
    digitalWrite(6, LOW);
    digitalWrite(7, LOW);
    digitalWrite(8, LOW);
}

void moveForward() {
    digitalWrite(5, HIGH);
    digitalWrite(6, LOW);
    digitalWrite(7, LOW);
    digitalWrite(8, HIGH);
}

void turnLeft() {
    digitalWrite(5, LOW);
    digitalWrite(6, LOW);
    digitalWrite(7, LOW);
    digitalWrite(8, HIGH);
}

void turnRight() {
    digitalWrite(5, HIGH);
    digitalWrite(6, LOW);
    digitalWrite(7, LOW);
    digitalWrite(8, LOW);
}

void lineFollow() {
    r1 = digitalRead(A1);
    r2 = digitalRead(A3);

    // both on line — go straight
    if (r1 == LOW && r2 == LOW) {
        moveForward();
    }
    // left sensor on line, right off — turn right to correct
    else if (r1 == LOW && r2 == HIGH) {
        turnRight();
    }
    // right sensor on line, left off — turn left to correct
    else if (r1 == HIGH && r2 == LOW) {
        turnLeft();
    }
    // both off line — continue last direction briefly instead of stopping
    else {
        moveForward();
        delay(50);
    }
}

void detectObstacle() {
    digitalWrite(trigPin, LOW);
    delayMicroseconds(2);
    digitalWrite(trigPin, HIGH);
    delayMicroseconds(10);
    digitalWrite(trigPin, LOW);
    duration = pulseIn(echoPin, HIGH, 30000);
    distance = (duration * 0.034) / 2;
    if (distance == 0 || distance > 200) distance = 200;
}

void leftBlind() {
    r2 = digitalRead(A3);
    if (r2 == LOW) {
        digitalWrite(5, LOW);
        digitalWrite(6, LOW);
        digitalWrite(7, HIGH);
        digitalWrite(8, LOW);
    }
    else {
        moveForward();
    }
}

void parkingLogic() {
    ElapsedTimeG = millis() - StartTimeG;
    if (ElapsedTimeG < 2500) {
        leftBlind();
    }
    else if (ElapsedTimeG < 6500) {
        lineFollow();
    }
    else {
        stopBuggy();
        Serial.println("Buggy Parked");
        delay(2000);
        flag = -1;
    }
}

void loop() {
    while (Serial.available() > 0) {
        char s = Serial.read();
        if (s == 'G' && flag != -1) {
            flag = 1;
            StartTimeG = millis();   // start timer for lap timing
        }
        else if (s == 'S') {
            flag = 0;
            stopBuggy();
        }
        else if (s == 'P' && flag == 1) {
            flag = 3;                // manually trigger parking via serial
            parkTriggerTime = millis();
            StartTimeG = millis();
            Serial.println("Parking triggered");
        }
    }

    if (flag == 1) {
        // trigger parking automatically after 30 seconds (approx 2 laps)
        if (millis() - StartTimeG > 30000) {
            flag = 3;
            parkTriggerTime = millis();
            StartTimeG = millis();
            Serial.println("Auto parking triggered");
        }
        else {
            detectObstacle();
            if (distance < 15 && distance > 0) {
                stopBuggy();
                obstacleClear = false;
            }
            else {
                if (!obstacleClear) {
                    delay(100);
                    obstacleClear = true;
                }
                lineFollow();
            }
        }
    }

    if (flag == 3) {
        parkingLogic();
    }
}
